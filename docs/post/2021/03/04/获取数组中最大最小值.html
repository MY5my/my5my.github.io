获取数组中最大最小值

平常

function smallest(array){                           
  return Math.min.apply(Math, array);             
}                                                 
function largest(array){                            
  return Math.max.apply(Math, array);             
}  
smallest([0, 1, 2.2, 3.3]); // 0
largest([0, 1, 2.2, 3.3]); // 3.3

es6语法

let list = [1, 2, 3, 4, 5]
Math.max(...list) // 5
Math.min(...list) // 1
