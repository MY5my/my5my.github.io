## 单个数组去重
```shell 
function unique(arr) {
    var arr1 = [];  
    for (var i = 0, len = arr.length; i < len; i++) {
        console.log(i)
        if (arr1.indexOf(arr[i]) === -1) {
            arr1.push(arr[i]);
        }
    }
    return arr1;
}
```
